1function [ alpha,eta,updater ] = BWalphaNloop( alpha,eta,updater )
 2%[alpha,eta]=BWALPHANLOOP(alpha,eta,updater) normalised forward variables for Baum-Welch algorithm
 3%   alpha    = normalised forward variables
 4%   eta      = normalisation factor
 5%   updater  = updater matrices (M*outProj*eta)
 6
 7siz=size(alpha,2)*[1 1];
 8
 9for i=2:length(eta)
10    alpha(i,:) = alpha(i-1,:) * reshape(updater(:,:,i-1),siz);
11    eta(i) = 1 / sum(alpha(i,:));
12    alpha(i,:) = alpha(i,:) * eta(i);
13    updater(:,:,i-1)=updater(:,:,i-1)*eta(i);
14end
15    
16end
